## 1240. 노드사이의 거리

<br>

#### 001. 첫 번째 풀이

> 두 노드 사이의 거리를 DFS를 활용하여 계산하였습니다.

<br>

```python
import sys


def get_dist(start, distance=0):
    global ans, e

    if start == e:
        ans = distance
        return

    if not graph[start]:
        return

    for nxt, dist in graph[start]:
        if not visited[nxt]:
            visited[nxt] = 1
            get_dist(nxt, distance+dist)
            visited[nxt] = 0


n, m = map(int, sys.stdin.readline().split())
graph = [[] for _ in range(n+1)]
for _ in range(n-1):
    s, e, dist = map(int, sys.stdin.readline().split())
    graph[s].append((e, dist))
    graph[e].append((s, dist))

for _ in range(m):
    s, e = map(int, sys.stdin.readline().split())

    visited = [0] * (n + 1)
    visited[s] = 1
    ans = 0
    get_dist(s)

    print(ans)
```

#### 002. 두 번째 풀이

> 두 노드 사이의 거리를 BFS를 활용하여 계산하였습니다.

<br>

```python
import sys
from collections import deque


def get_dist(start, end):
    visited = [0] * (n + 1)
    visited[start] = 1

    q = deque([(start, 0)])
    while q:
        s, distance = q.popleft()

        if s == end:
            return distance

        for nxt, dist in graph[s]:
            if not visited[nxt]:
                visited[nxt] = 1
                q.append((nxt, distance+dist))


n, m = map(int, sys.stdin.readline().split())
graph = [[] for _ in range(n+1)]
for _ in range(n-1):
    s, e, dist = map(int, sys.stdin.readline().split())
    graph[s].append((e, dist))
    graph[e].append((s, dist))

for _ in range(m):
    s, e = map(int, sys.stdin.readline().split())
    print(get_dist(s, e))
```
